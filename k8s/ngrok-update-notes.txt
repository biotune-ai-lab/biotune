apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: cert-manager-webhook
  namespace: cert-manager
spec:
  ingressClassName: ngrok
  rules:
  - host: cert-manager-webhook-55i2b9dq.ngrok.app
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: cert-manager-webhook
            port:
              number: 443
--------------------------------------------------------------------------------
Applying this manifest will make the service "cert-manager-webhook"
available on the public internet at "https://cert-manager-webhook-55i2b9dq.ngrok.app/".
